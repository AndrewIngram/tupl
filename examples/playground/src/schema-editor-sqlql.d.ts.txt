declare module "sqlql" {
  export type SqlScalarType =
    | "text"
    | "integer"
    | "real"
    | "blob"
    | "boolean"
    | "timestamp"
    | "date"
    | "datetime"
    | "json";

  export interface DataEntityColumnMetadata<TRead = unknown> {
    source: string;
    type?: SqlScalarType;
    nullable?: boolean;
    primaryKey?: boolean;
    unique?: boolean;
    enum?: readonly string[];
    physicalType?: string;
    physicalDialect?: "postgres" | "sqlite";
    readonly __read__?: TRead;
  }

  export interface ColumnDefinition {
    type: SqlScalarType;
    nullable?: boolean;
    primaryKey?: boolean;
    unique?: boolean;
    enum?: readonly string[];
    enumFrom?: string | SchemaColRefToken;
    enumMap?: Record<string, string>;
    physicalType?: string;
    physicalDialect?: "postgres" | "sqlite";
    foreignKey?: ColumnForeignKeyReference;
    description?: string;
  }

  export type TableColumnDefinition = SqlScalarType | ColumnDefinition;

  export interface TableDefinition {
    provider?: string;
    columns: Record<string, TableColumnDefinition>;
  }

  export interface SchemaDefinition {
    tables: Record<string, TableDefinition>;
  }

  export interface DataEntityHandle<
    TColumns extends string = string,
    TRow extends Partial<Record<TColumns, unknown>> = Record<TColumns, unknown>,
    TColumnMetadata extends Partial<Record<TColumns, DataEntityColumnMetadata<any>>> = {
      [K in TColumns]: DataEntityColumnMetadata<K extends keyof TRow ? TRow[K] : unknown>;
    },
  > {
    kind: "data_entity";
    entity: string;
    provider: string;
    columns?: {
      [K in TColumns]: K extends keyof TColumnMetadata
        ? TColumnMetadata[K] & {
            readonly __read__?: K extends keyof TRow ? TRow[K] : unknown;
          }
        : DataEntityColumnMetadata<K extends keyof TRow ? TRow[K] : unknown>;
    };
    readonly __columns__?: TColumns;
  }

  export interface ExecutableSchema<TContext, TSchema extends SchemaDefinition = SchemaDefinition> {
    schema: TSchema;
    query(input: { context: TContext; sql: string }): Promise<Array<Record<string, unknown>>>;
    createSession(input: {
      context: TContext;
      sql: string;
      options?: Record<string, unknown>;
    }): unknown;
    explain?(input: { context: TContext; sql: string }): Promise<unknown>;
  }

  export interface SchemaDslTableToken<TColumns extends string = string> {
    kind: "dsl_table_token";
    readonly __id: symbol;
    readonly __brand?: TColumns;
  }

  export interface SchemaColRefToken {
    kind: "dsl_col_ref";
    ref?: string;
    column?: string;
  }

  export interface SchemaColumnLensDefinition {
    source: string | SchemaColRefToken;
    type?: SqlScalarType;
    nullable?: boolean;
    primaryKey?: boolean;
    unique?: boolean;
    enum?: readonly string[];
    enumFrom?: string | SchemaColRefToken;
    enumMap?: Record<string, string>;
    physicalType?: string;
    physicalDialect?: "postgres" | "sqlite";
    foreignKey?: ColumnForeignKeyReference;
    description?: string;
    coerce?: "isoTimestamp" | ((value: unknown) => unknown);
  }

  export interface ColumnForeignKeyReference {
    table: string;
    column: string;
    name?: string;
    onDelete?: "NO ACTION" | "RESTRICT" | "CASCADE" | "SET NULL" | "SET DEFAULT";
    onUpdate?: "NO ACTION" | "RESTRICT" | "CASCADE" | "SET NULL" | "SET DEFAULT";
  }

  export interface SchemaViewEqExpr {
    kind: "eq";
    left: SchemaColRefToken;
    right: SchemaColRefToken;
  }

  export interface SchemaViewAggregateMetric {
    kind: "metric";
    fn: "count" | "sum" | "avg" | "min" | "max";
    column?: SchemaColRefToken;
    distinct?: boolean;
  }

  export type SchemaViewRelNodeInput = unknown;

  interface TypedColumnOptions {
    nullable?: boolean;
    primaryKey?: boolean;
    unique?: boolean;
    enum?: readonly string[];
    enumFrom?: string | SchemaColRefToken;
    enumMap?: Record<string, string>;
    physicalType?: string;
    physicalDialect?: "postgres" | "sqlite";
    foreignKey?: ColumnForeignKeyReference;
    description?: string;
    coerce?: "isoTimestamp" | ((value: unknown) => unknown);
  }

  type StripNullish<T> = Exclude<T, null | undefined>;
  type IsAny<T> = 0 extends 1 & T ? true : false;
  type IsExactly<T, U> = [T] extends [U] ? ([U] extends [T] ? true : false) : false;
  type CompatibleSourceScalarType<TTarget extends SqlScalarType> = TTarget extends "real"
    ? "real" | "integer"
    : TTarget;
  type InferSourceScalarTypeFromRead<TRead> = IsAny<TRead> extends true
    ? never
    : [StripNullish<TRead>] extends [never]
      ? never
      : StripNullish<TRead> extends number
        ? "integer" | "real"
        : StripNullish<TRead> extends boolean
          ? "boolean"
          : StripNullish<TRead> extends Uint8Array
            ? "blob"
            : StripNullish<TRead> extends Date
              ? "timestamp"
              : StripNullish<TRead> extends string
                ? "text"
                : StripNullish<TRead> extends object
                  ? "json"
                  : never;
  type ExplicitSourceScalarType<TMetadata> = Extract<
    TMetadata extends { type?: infer TType } ? TType : never,
    SqlScalarType
  >;
  type KnownSourceScalarType<TMetadata> = [ExplicitSourceScalarType<TMetadata>] extends [never]
    ? InferSourceScalarTypeFromRead<TMetadata extends { readonly __read__?: infer TRead } ? TRead : never>
    : IsExactly<ExplicitSourceScalarType<TMetadata>, SqlScalarType> extends true
      ? InferSourceScalarTypeFromRead<TMetadata extends { readonly __read__?: infer TRead } ? TRead : never>
      : ExplicitSourceScalarType<TMetadata>;
  type IsCompileTimeCompatibleSourceType<
    TSource extends SqlScalarType,
    TTarget extends SqlScalarType,
  > = [Extract<TSource, CompatibleSourceScalarType<TTarget>>] extends [never] ? false : true;
  type CompatibleColumnName<
    TColumns extends string,
    TColumnMetadata extends Partial<Record<TColumns, DataEntityColumnMetadata<any>>>,
    TTarget extends SqlScalarType,
  > = Extract<
    {
      [K in TColumns]:
        [KnownSourceScalarType<TColumnMetadata[K]>] extends [never]
          ? K
          : IsCompileTimeCompatibleSourceType<KnownSourceScalarType<TColumnMetadata[K]>, TTarget> extends true
            ? K
            : never;
    }[TColumns],
    string
  >;

  interface SchemaTypedColumnDefinition<TSourceColumn extends string = string> {
    kind: "dsl_typed_column";
    sourceColumn: TSourceColumn;
    definition: TableColumnDefinition;
    coerce?: "isoTimestamp" | ((value: unknown) => unknown);
  }

  interface SchemaTypedColumnBuilder<
    TColumns extends string,
    TColumnMetadata extends Partial<Record<TColumns, DataEntityColumnMetadata<any>>> = {
      [K in TColumns]: DataEntityColumnMetadata<unknown>;
    },
  > {
    id: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "text">>(
        column: TColumn,
        options?: Omit<TypedColumnOptions, "primaryKey" | "nullable" | "enum" | "enumFrom" | "enumMap">,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: Omit<TypedColumnOptions, "primaryKey" | "nullable" | "enum" | "enumFrom" | "enumMap"> & {
          coerce: "isoTimestamp" | ((value: unknown) => unknown);
        },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    string: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "text">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    integer: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "integer">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    real: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "real">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    blob: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "blob">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    boolean: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "boolean">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    timestamp: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "timestamp">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    date: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "date">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    datetime: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "datetime">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
    json: {
      <TColumn extends CompatibleColumnName<TColumns, TColumnMetadata, "json">>(
        column: TColumn,
        options?: TypedColumnOptions,
      ): SchemaTypedColumnDefinition<TColumn>;
      <TColumn extends TColumns>(
        column: TColumn,
        options: TypedColumnOptions & { coerce: "isoTimestamp" | ((value: unknown) => unknown) },
      ): SchemaTypedColumnDefinition<TColumn>;
    };
  }

  interface SchemaDslRelHelpers {
    scan<TColumns extends string>(
      table: SchemaDslTableToken<TColumns> | { tableToken: SchemaDslTableToken<TColumns> },
    ): SchemaViewRelNodeInput;
    join(input: {
      left: SchemaViewRelNodeInput;
      right: SchemaViewRelNodeInput;
      on: SchemaViewEqExpr;
      type?: "inner" | "left" | "right" | "full";
    }): SchemaViewRelNodeInput;
    aggregate<TGroupBy extends Record<string, SchemaColRefToken>, TMeasures extends Record<string, SchemaViewAggregateMetric>>(input: {
      from: SchemaViewRelNodeInput;
      groupBy: TGroupBy;
      measures: TMeasures;
    }): SchemaViewRelNodeInput;
  }

  export interface SchemaDslHelpers<TContext> {
    table: {
      <
        TSourceColumns extends string,
        TColumns extends string,
        TRow extends Partial<Record<TSourceColumns, unknown>> = Record<TSourceColumns, unknown>,
        TColumnMetadata extends Partial<Record<TSourceColumns, DataEntityColumnMetadata<any>>> = {
          [K in TSourceColumns]: DataEntityColumnMetadata<K extends keyof TRow ? TRow[K] : unknown>;
        },
      >(
        from: DataEntityHandle<TSourceColumns, TRow, TColumnMetadata>,
        input: {
          columns: (helpers: { col: SchemaTypedColumnBuilder<TSourceColumns, TColumnMetadata> }) => Record<TColumns, unknown>;
          constraints?: Record<string, unknown>;
        },
      ): {
        kind: "dsl_table";
        tableToken: SchemaDslTableToken<TColumns>;
      };
    };
    view: <TRelColumns extends string, TColumns extends string>(input: {
      rel: (helpers: SchemaDslRelHelpers & {
        col: {
          <TCols extends string, TColumn extends TCols>(
            table: SchemaDslTableToken<TCols> | { tableToken: SchemaDslTableToken<TCols> },
            column: TColumn,
          ): SchemaColRefToken;
        };
        expr: {
          eq: (left: SchemaColRefToken, right: SchemaColRefToken) => SchemaViewEqExpr;
        };
        agg: {
          countDistinct: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          sum: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          count: () => SchemaViewAggregateMetric;
          sumDistinct: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          avg: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          avgDistinct: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          min: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
          max: (column: SchemaColRefToken) => SchemaViewAggregateMetric;
        };
      }, context: TContext) => SchemaViewRelNodeInput;
      columns:
        | ((helpers: {
            col: SchemaTypedColumnBuilder<
              TRelColumns,
              { [K in TRelColumns]: DataEntityColumnMetadata<unknown> }
            >;
          }) => Record<TColumns, unknown>)
        | Record<
            TColumns,
            SchemaColumnLensDefinition | SchemaColRefToken | SchemaTypedColumnDefinition<TRelColumns>
          >;
      constraints?: Record<string, unknown>;
    }) => unknown;
  }

  export function createDataEntityHandle<TColumns extends string = string>(input: {
    entity: string;
    provider: string;
  }): DataEntityHandle<TColumns>;

  export function defineSchema<TContext>(
    schemaBuilder: (helpers: SchemaDslHelpers<TContext>) => {
      tables: Record<string, unknown>;
    },
  ): SchemaDefinition;
  export function defineSchema<TSchema extends SchemaDefinition>(schema: TSchema): TSchema;
  export function createExecutableSchema<TContext>(
    schemaBuilder: (helpers: SchemaDslHelpers<TContext>) => {
      tables: Record<string, unknown>;
    },
  ): ExecutableSchema<TContext>;
}

    };
  };
}

declare module "drizzle-orm/pg-core" {
  interface PgColumn<TName extends string = string> {
    readonly name: TName;
  }

  interface ColumnBuilder<TName extends string = string> {
    build(): PgColumn<TName>;
    primaryKey(): ColumnBuilder<TName>;
    notNull(): ColumnBuilder<TName>;
    references(ref: () => unknown): ColumnBuilder<TName>;
  }

  type BuiltColumns<TColumns extends Record<string, ColumnBuilder<string>>> = {
    [K in keyof TColumns]: TColumns[K] extends ColumnBuilder<infer TName> ? PgColumn<TName> : never;
  };

  export function pgTable<
    TTableName extends string,
    TColumns extends Record<string, ColumnBuilder<string>>,
  >(name: TTableName, columns: TColumns): BuiltColumns<TColumns> & {
    _: {
      config: {
        name: TTableName;
        dialect: "postgres";
      };
    };
  };
  export function text<TName extends string>(name: TName): ColumnBuilder<TName>;
  export function integer<TName extends string>(name: TName): ColumnBuilder<TName>;
  export function boolean<TName extends string>(name: TName): ColumnBuilder<TName>;
  export function timestamp<TName extends string>(name: TName, options?: { mode?: string }): ColumnBuilder<TName>;
}
